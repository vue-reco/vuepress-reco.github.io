Multiple code themes

reco_luan 2020-07-05

# Theme style

Go to prismjs.

# Set the theme

// .vuepress/config.js

module.exports = {
  theme: 'reco',
  themeConfig: {
    /**
     * support for
     * 'default'
     * 'funky'
     * 'okaidia'
     * 'solarizedlight'
     * 'tomorrow'
     */
    codeTheme: 'tomorrow' // default 'tomorrow'
  }
}
