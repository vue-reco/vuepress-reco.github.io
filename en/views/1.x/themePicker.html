Theme Picker

reco_luan 2019-08-23

# Disable

// .vuepress/config.js
module.exports = {
  theme: 'reco',
  themeConfig: {
    themePicker: false
  }  
}

# Customize Color

// .vuepress/config.js
module.exports = {
  theme: 'reco',
  themeConfig: {
    themePicker: {
      colorName1: 'red',
      colorName2: 'yellow',
      colorName3: 'blue'
    }
  }  
}  

// .vuepress/styles/palette.styl
$themePicker = {
  colorName1: red,
  colorName2: yellow,
  colorName3: blue
}
